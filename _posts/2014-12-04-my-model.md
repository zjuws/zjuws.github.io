#define N 100010
using namespace std;
int Tree[N];
int num[N];
int f[N];
int findRoot(int x){
	if(Tree[x]==x) return x;
	int fx=Tree[x];
	Tree[x]=findRoot(Tree[x]);
	f[x]=f[fx]+f[x];
	return Tree[x];
}
void merge(int x,int y) {
    int fx = findRoot(x);
    int fy = findRoot(y);
    if(fx==fy) {

    }else {
        Tree[fx] = fy;
        f[fx] = (f[y] - f[x] + 1);
    }
}
int main(){
	int n,s;
	double p,r;
	while(~scanf("%d%lf%lf",&n,&p,&r)){
		for(int i=0;i<n;i++){
			Tree[i]=i;
			f[i]=0;
		}
		for(int i=0;i<n;i++){
			scanf("%d",&s);
			if(s==-1) continue;
			merge(i,s);
		}

		int sameN=0;
		int deep=0;
		for(int i=0;i<n;i++){
			findRoot(i);
			//printf("%d: %d\n",i,f[i]);

				deep=f[i];
				sameN=1;

				sameN++;
